import React, { useState } from 'react';
import { GoogleGenAI, Type } from '@google/genai';
import { Sparkles, Loader2, Check } from 'lucide-react';
import { motion, AnimatePresence } from 'framer-motion';

const AICardRecommender: React.FC = () => {
  const [profile, setProfile] = useState('');
  const [recommendation, setRecommendation] = useState<any>(null);
  const [loading, setLoading] = useState(false);

  const getRecommendation = async () => {
    if (!profile.trim()) return;
    setLoading(true);
    try {
      const ai = new GoogleGenAI({ apiKey: process.env.API_KEY });
      const response = await ai.models.generateContent({
        model: 'gemini-3-flash-preview',
        contents: `Analise o seguinte perfil de usuário para recomendar um dos nossos planos de cartão cripto (Essential, Pro, ou Black Elite).
        
        Perfil do Usuário: "${profile}"
        
        Regras de Planos:
        1. Essential: Para iniciantes, 1% cashback, limite de $1k/mês.
        2. Pro: Para investidores frequentes, 2.5% cashback, benefícios em aeroportos, limite de $10k/mês.
        3. Black Elite: Para baleias/high net worth, 5% cashback, concierge 24h, limites ilimitados.

        Retorne um JSON com: planName, reason (curta em PT-BR), highlightBenefit.`,
        config: {
          responseMimeType: "application/json",
          responseSchema: {
            type: Type.OBJECT,
            properties: {
              planName: { type: Type.STRING },
              reason: { type: Type.STRING },
              highlightBenefit: { type: Type.STRING }
            },
            required: ["planName", "reason", "highlightBenefit"]
          }
        }
      });

      const data = JSON.parse(response.text);
      setRecommendation(data);
    } catch (error) {
      console.error(error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <section className="py-24 bg-white">
      <div className="max-w-4xl mx-auto px-4 sm:px-6 lg:px-8">
        <motion.div 
          initial={{ opacity: 0, scale: 0.95 }}
          whileInView={{ opacity: 1, scale: 1 }}
          viewport={{ once: true }}
          className="bg-slate-50 rounded-[3rem] p-10 lg:p-16 border border-slate-100"
        >
          <div className="flex flex-col items-center text-center gap-4 mb-12">
            <div className="w-16 h-16 bg-black rounded-2xl flex items-center justify-center text-accent">
              <Sparkles size={32} />
            </div>
            <div>
              <h2 className="text-3xl font-black text-black uppercase tracking-tighter">Match Inteligente</h2>
              <p className="text-slate-500 font-medium">Nossa IA escolhe o melhor nível de privilégio para você.</p>
            </div>
          </div>

          <div className="space-y-8">
            <div className="relative">
              <label className="block text-xs font-black text-slate-400 uppercase tracking-widest mb-3 ml-2">
                Conte-nos sobre sua rotina financeira
              </label>
              <textarea
                value={profile}
                onChange={(e) => setProfile(e.target.value)}
                placeholder="Ex: Viajo muito a trabalho, invisto em ETH e gasto cerca de 8k por mês em restaurantes."
                className="w-full h-40 px-8 py-6 rounded-3xl border-2 border-slate-200 bg-white focus:ring-4 focus:ring-accent/20 focus:border-black transition-all outline-none text-black font-medium resize-none text-lg shadow-sm"
              />
            </div>

            <button
              onClick={getRecommendation}
              disabled={loading || !profile.trim()}
              className="w-full bg-black text-white py-6 rounded-3xl font-black text-xl flex items-center justify-center gap-3 hover:bg-accent hover:text-black transition-all disabled:opacity-50 active:scale-[0.98]"
            >
              {loading ? (
                <Loader2 className="animate-spin" />
              ) : (
                <>Analisar Perfil <Sparkles size={22} /></>
              )}
            </button>

            <AnimatePresence mode="wait">
              {recommendation && (
                <motion.div 
                  initial={{ opacity: 0, y: 20 }}
                  animate={{ opacity: 1, y: 0 }}
                  exit={{ opacity: 0, y: -20 }}
                  className="mt-12 p-8 bg-white rounded-3xl border-2 border-accent shadow-xl shadow-accent/10"
                >
                  <div className="flex justify-between items-start mb-6">
                    <span className="text-xs font-black uppercase tracking-[0.2em] text-black px-4 py-1.5 bg-accent rounded-full">
                      Resultado da Análise
                    </span>
                    <div className="flex items-center text-lime-600 text-sm font-black uppercase tracking-tighter">
                      <Check size={18} className="mr-1" /> Perfeito para você
                    </div>
                  </div>
                  <h3 className="text-4xl font-black text-black mb-4 tracking-tighter">{recommendation.planName}</h3>
                  <p className="text-slate-600 mb-8 text-lg font-medium leading-relaxed">{recommendation.reason}</p>
                  <div className="flex items-center gap-3 text-black font-black bg-accent/30 p-5 rounded-2xl border border-accent/40">
                    <Sparkles size={20} />
                    <span>Destaque: {recommendation.highlightBenefit}</span>
                  </div>
                </motion.div>
              )}
            </AnimatePresence>
          </div>
        </motion.div>
      </div>
    </section>
  );
};

export default AICardRecommender;
